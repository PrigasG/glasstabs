\section{Setup}

Call once in every UI:
\begin{Verbatim}
library(glasstabs)
ui <- fluidPage(
  \fn{useGlassTabs},
  # widgets go here
)
\end{Verbatim}

Run a built-in demo:
\begin{Verbatim}
\fn{runGlassExample}         # list all
runGlassExample("smoke-test")
runGlassExample("basic")
runGlassExample("bs4dash")
\end{Verbatim}

% ══════════════════════════════════════════════════════
\section{Tab widget — UI}

\begin{Verbatim}
glassTabsUI(
  id,            # module id
  ...,           # glassTabPanel() calls
  selected=NULL, # initial tab value
  wrap=TRUE,
  extra_ui=NULL, # right-side widget
  theme="dark"   # or "light" / theme obj
)

glassTabPanel(
  value,         # unique tab id
  label,         # button text
  ...,           # pane content
  icon=NULL,     # shiny::icon() etc.
  selected=FALSE
)
\end{Verbatim}

\textbf{Minimal example:}
\begin{Verbatim}
glassTabsUI(
  "main",
  glassTabPanel("a","Alpha",selected=TRUE,
    p("Alpha content")),
  glassTabPanel("b","Beta",
    icon=icon("star"),
    p("Beta content"))
)
\end{Verbatim}

% ══════════════════════════════════════════════════════
\section{Tab widget — server}

\begin{Verbatim}
server <- function(input, output, session) {
  # Returns reactive active tab value
  active <- glassTabsServer("main")
  # With bookmark support (default TRUE)
  active <- glassTabsServer("main",
              bookmark=TRUE)
  observe(print(active()))
}
\end{Verbatim}

\textbf{Module pattern:}\\
UI: \code{glassTabsUI(\textbf{ns("tabs")}, ...)}\\
Server: \code{glassTabsServer(\textbf{"tabs"})} \\
{\footnotesize\color{gtgray}Pass bare id, not \code{ns("tabs")}, to server.}

% ══════════════════════════════════════════════════════
\section{Tab server actions}

\begin{Verbatim}
# Switch active tab
updateGlassTabsUI(session,"main","b")

# Show / hide
showGlassTab(session,"main","admin")
hideGlassTab(session,"main","admin")

# Disable / enable (stays visible)
disableGlassTab(session,"main","beta")
enableGlassTab(session,"main","beta")

# Badge counts (0 = hide badge)
updateGlassTabBadge(session,"main",
  "inbox", count=5L)
updateGlassTabBadge(session,"main",
  "inbox", count=0L)

# Add / remove at runtime
appendGlassTab(session,"main",
  glassTabPanel("new","New",p("...")),
  select=TRUE)
removeGlassTab(session,"main","new")
\end{Verbatim}

% ══════════════════════════════════════════════════════
\section{Dynamic tabs (renderGlassTabs)}

\begin{Verbatim}
# UI
glassTabsOutput("dyn_tabs")

# Server
output$dyn_tabs <- renderGlassTabs({
  panels <- lapply(my_data(), function(r)
    glassTabPanel(r$id, r$label, p(r$id)))
  do.call(glassTabsUI,
    c(list("dyn"), panels))
})
\end{Verbatim}

{\footnotesize\color{gtgray}JS re-initialises automatically after each render.}

\section{Bookmarking}

Enable in the app, then \fn{glassTabsServer} handles save/restore automatically:
\begin{Verbatim}
shinyApp(ui, server,
  enableBookmarking="url")
\end{Verbatim}
Opt out per widget: \code{bookmark=FALSE}.

\begin{Verbatim}

  )

\begin{itemize}

  \item \arg{content\_bg}, \arg{content\_border}
  \item \arg{card\_bg}, \arg{card\_text}
\end{itemize}

\section{Multi-select — server}

\begin{Verbatim}
# Reactive list: selected() and style()
vals <- glassMultiSelectValue(input,"m")
vals$selected()
vals$style()     # "checkbox"|"filled"|"dot"

# Update from server
updateGlassMultiSelect(session,"m",
  selected=c("rev","ord"),
  check_style="filled")
# Clear all
updateGlassMultiSelect(session,"m",
  selected=character(0))

# Filter-tag display
glassFilterTags("m")  # in UI

{\footnotesize
\begin{tabular}{@{}ll@{}}
\textbf{Widget} & \textbf{Input key} \\
\hline
tabs            & \code{input[["id-active\_tab"]]} \\
multi-select    & \code{input\$inputId} \\
multi style     & \code{input\$inputId\_style} \\

\begin{itemize}
  \item \fn{glassTabsServer} → \code{active()}

\section{Common gotchas}

\begin{itemize}
  \item \fn{useGlassTabs} must appear once in UI.
  \item In modules: \code{ns("tabs")} in UI, bare \code{"tabs"} in server.
  \item Duplicate \code{glassTabPanel()} values → error.
  \item \arg{selected} = tab \textbf{value}, not label.
  \item \fn{glassSelectValue} returns a reactive, not a list.
  \item \fn{glassMultiSelectValue} returns \code{list(selected(), style())}.
  \item \code{selected} in multi-select uses \textbf{values}, not labels.
  \item After \code{load\_all()}, reload the browser to pick up JS changes.
\end{itemize}
