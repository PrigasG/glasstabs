\mysection{Setup}

Call once in every UI:
\begin{lstlisting}
library(glasstabs)
ui <- fluidPage(
  useGlassTabs(),
  # widgets go here
)
\end{lstlisting}

Run a built-in demo:
\begin{lstlisting}
runGlassExample()       # list all
runGlassExample("smoke-test")
runGlassExample("basic")
runGlassExample("bs4dash")
\end{lstlisting}

% ══════════════════════════════════════════════════════
\mysection{Tab widget --- UI}

\begin{lstlisting}
glassTabsUI(
  id,             # module id
  ...,            # glassTabPanel()s
  selected = NULL,# initial tab value
  wrap     = TRUE,
  extra_ui = NULL,# right-side widget
  theme    = "dark" # or "light"
)

glassTabPanel(
  value,          # unique tab id
  label,          # button text
  ...,            # pane content
  icon = NULL,    # shiny::icon() etc.
  selected = FALSE
)
\end{lstlisting}

\textbf{Example:}
\begin{lstlisting}
glassTabsUI(
  "main",
  glassTabPanel("a", "Alpha",
    selected = TRUE,
    p("Alpha content")),
  glassTabPanel("b", "Beta",
    icon = icon("star"),
    p("Beta content"))
)
\end{lstlisting}

% ══════════════════════════════════════════════════════
\mysection{Tab widget --- server}

\begin{lstlisting}
server <- function(input, output, session) {
  # Reactive active tab value
  active <- glassTabsServer("main")
  # With bookmarking (default TRUE)
  active <- glassTabsServer("main",
              bookmark = TRUE)
  observe(print(active()))
}
\end{lstlisting}

\textbf{Module pattern:}\\
UI:~\code{glassTabsUI(\textbf{ns("tabs")}, ...)}\\
Server:~\code{glassTabsServer(\textbf{"tabs"})}\\
{\tiny\color{gtgray}Pass bare id to server --- not \code{ns("tabs")}.}

\mysection{Tab server actions}

\begin{lstlisting}
# Switch active tab
updateGlassTabsUI(session, "main", "b")

# Show / hide
showGlassTab(session, "main", "admin")
hideGlassTab(session, "main", "admin")

# Disable / enable (stays visible)
disableGlassTab(session, "main", "b")
enableGlassTab(session,  "main", "b")

# Badge counts  (0 or NA = hide)
updateGlassTabBadge(session, "main",
  "inbox", count = 5L)
updateGlassTabBadge(session, "main",
  "inbox", count = 0L)

# Add / remove at runtime
appendGlassTab(session, "main",
  glassTabPanel("new", "New", p("...")),
  select = TRUE)
removeGlassTab(session, "main", "new")
\end{lstlisting}

\mysection{Dynamic tabs}

\begin{lstlisting}
# UI placeholder
glassTabsOutput("dyn_tabs")

# Server render
output$dyn_tabs <- renderGlassTabs({
  panels <- lapply(my_data(),
    function(r)
      glassTabPanel(r$id, r$label,
        p(r$id)))
  do.call(glassTabsUI,
    c(list("dyn"), panels))
})
\end{lstlisting}
{\tiny\color{gtgray}JS re-initialises automatically after each render.}

\mysection{Bookmarking}

Enable on the app, \gtfn{glassTabsServer} handles save/restore:
\begin{lstlisting}
shinyApp(ui, server,
  enableBookmarking = "url")
\end{lstlisting}

Opt out per widget:
\begin{lstlisting}
glassTabsServer("tabs",
  bookmark = FALSE)
\end{lstlisting}

\mysection{Common gotchas}

\begin{itemize}
  \item \gtfn{useGlassTabs} once in UI.
  \item Modules: \code{ns("tabs")} in UI, bare \code{"tabs"} in server.
  \item Duplicate \code{glassTabPanel()} values $\to$ error.
  \item \gtarg{selected} = tab \textbf{value}, not label.
  \item \gtfn{glassSelectValue} returns reactive, not list.
  \item \gtfn{glassMultiSelectValue} returns
        \code{list(selected(), style())}.
  \item Multi-select \gtarg{selected} uses values, not labels.
  \item After \code{load\_all()}, reload browser to pick up JS changes.
\end{itemize}
